function getMeetingYear(dateString) { const date = new Date(dateString); return date.getFullYear(); } const FALLBACK = 'https://kellybearclawz.github.io/bookclub/default-cover.jpg'; // Fetch covers in batches of 20 ISBNs at a time to avoid URL length limits // Uses Google Dynamic Links API (no quota, single request per batch) function fetchCoverBatch(isbns) { return new Promise((resolve) => { const result = {}; if (!isbns.length) { resolve(result); return; } const bibkeys = isbns.map(i => `ISBN:${i}`).join(','); const cbName = `__gbcb_${Date.now()}`; const timeout = setTimeout(() => { delete window[cbName]; resolve(result); }, 8000); window[cbName] = function(data) { clearTimeout(timeout); for (const key in data) { const isbn = key.replace('ISBN:', ''); if (data[key].thumbnail_url) { result[isbn] = data[key].thumbnail_url.replace('http://', 'https://'); } } delete window[cbName]; resolve(result); }; const script = document.createElement('script'); script.src = `https://books.google.com/books?bibkeys=${encodeURIComponent(bibkeys)}&jscmd=viewapi&callback=${cbName}`; script.onerror = () => { clearTimeout(timeout); resolve(result); }; document.head.appendChild(script); }); } async function fetchCoversForISBNs(isbns) { const unique = [...new Set(isbns.filter(Boolean))]; const BATCH_SIZE = 20; const coverMap = {}; for (let i = 0; i < unique.length; i += BATCH_SIZE) { const batch = unique.slice(i, i + BATCH_SIZE); const batchResult = await fetchCoverBatch(batch); Object.assign(coverMap, batchResult); } return coverMap; } async function renderBooks(data) { const shelf = document.getElementById('bookshelf'); // Group by year const booksByYear = {}; for (const book of data) { const year = new Date(book['Meeting Date']).getFullYear(); if (!booksByYear[year]) booksByYear[year] = []; booksByYear[year].push(book); } const years = Object.keys(booksByYear).sort().reverse(); // Year jump links const yearLinksDiv = document.createElement('div'); yearLinksDiv.className = 'year-links'; yearLinksDiv.innerHTML = years.map(y => `${y}`).join(''); shelf.appendChild(yearLinksDiv); // Fetch all covers in batches before rendering const allISBNs = data.map(b => b.ISBN?.replace(/[^0-9Xx]/g, '')).filter(Boolean); const coverMap = await fetchCoversForISBNs(allISBNs); // Render each year section for (const year of years) { const section = document.createElement('section'); section.id = `year-${year}`; section.innerHTML = `

${year}

`; const bookContainer = document.createElement('div'); bookContainer.className = 'book-container'; const booksInYear = [...booksByYear[year]].reverse(); booksInYear.forEach((book, index) => { const isbn = book.ISBN?.replace(/[^0-9Xx]/g, ''); const coverUrl = (isbn && coverMap[isbn]) ? coverMap[isbn] : FALLBACK; const bookDiv = document.createElement('div'); bookDiv.className = 'book-card fade-in'; bookDiv.style.animationDelay = `${index * 0.06}s`; const img = document.createElement('img'); img.src = coverUrl; img.alt = `Cover of ${book.Title}`; img.onerror = () => { img.onerror = null; img.src = FALLBACK; }; const info = document.createElement('div'); info.innerHTML = ` ${book.Title}

Meeting: ${book['Meeting Date']}
